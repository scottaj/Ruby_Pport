        def initialize(address)

            @port = nil

            if RUBY_PALTFORM.match(/linux/i)
                @port = LinuxPport.new(address)
            elsif RUBY_PLATFORM.match(/mingw/i) or RUBY_PLATFORM.match(/mswin/i)
                @port = WinPport.new(address)
            end
        end
